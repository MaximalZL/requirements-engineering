"Пользователи"
-
ID PK int
Email string UNIQUE
Пароль string
Имя string
Фамилия NULL string
Отчество NULL string
Дата_рождения Date NULL
Пол string NULL
Дата_регистрации Date
Фото NULL string
Статус profile_status # Перечисление cтатусов профиля
ID_роли role_enym  # Перечисление ролей

"Контакты"
-
ID PK int
ID_пользователя int FK >- "Пользователи".ID
Тип contact_enym # Перечисление ролей
Контакт string

"Питомцы"
-
ID PK int
Имя string
Возраст int
Пол pet_gender_enym # Перечисление пола питомцев
Фото string
Вид string
Порода pet_breed_enym # Перечисление пород питомцев
ID_владельца int FK >- "Пользователи".ID

"Документы"
-
ID PK int
Тип string
Дата_выдачи Date
Дата_окончания NULL Date
Описание string
Фото string
ID_питомца int FK >- "Питомцы".ID

"Объявления"
-
ID PK int
ID_пользователя(Продавца) int FK >- "Пользователи".ID
ID_питомца UNIQUE int FK >- "Питомцы".ID
Описание string
Цена int
Дата_создания Date
Дата_закрытия NULL Date
Статус advertisement_status # Cтатус объявления о продаже
Метка MapType # Тип отметки на карте

"История сделок"
-
ID PK int
ID_объявления int FK >- "Объявления".ID
ID_пользователя(Продавец) int FK >- "Пользователи".ID
ID_пользователя(Покупатель) int FK >- "Пользователи".ID
Дата_сделки Date
Сумма int
Способ_оплаты string 
Статус_сделки payment_status # Cтатус оплаты

"Сообщения"
-
ID PK int
ID_отправителя int FK >- "Пользователи".ID  # Пользователь, отправивший сообщение
ID_получателя int FK >- "Пользователи".ID  # Пользователь, получивший сообщение
Текст string  # Содержимое сообщения
Дата_отправки DateTime  # Дата и время отправки сообщения
Статус message_status  # Перечисление статусов сообщения (например, "Прочитано", "Не прочитано")
ID_объявления NULL int FK >- "Объявления".ID  # (опционально) ID объявления, к которому относится сообщение
